7 5.0 Wrangling Data —-

bike_orderlines_wrangled <- bike_orderlines %>%
  separate(col    = location,
           into   = c("City", "State"), sep = ",") %>% mutate(total.price = price * quantity) %>%
set_names(names(.) %>% str_replace_all("\\.", "_"))

9 6.1 Sales by State —-

sales_by_state <- bike_orderlines_wrangled %>%
 
  # Select columns
  select(order_date, total_price, State) %>%

  # Add year column
  mutate(year = year(order_date)) %>%
 
  # Grouping by year and summarizing sales
  group_by(State) %>% 
  summarize(sales = sum(total_price)) %>%
 
  # Optional: Add a column that turns the numbers into a currency format 
  # (makes it in the plot optically more appealing)
  # mutate(sales_text = scales::dollar(sales)) <- Works for dollar values
  mutate(state_text = scales::dollar(sales, big.mark = ".", 
                                     decimal.mark = ",", 
                                     prefix = "", 
                                     suffix = " €"))

sales_by_year_state <- bike_orderlines_wrangled %>%
 
  # Select columns
  select(order_date, total_price, State) %>%

  # Add year column
  mutate(year = year(order_date)) %>%
 
  # Grouping by year and summarizing sales
  group_by(year, State) %>% 
  summarize(sales = sum(total_price)) %>%
 
  # Optional: Add a column that turns the numbers into a currency format 
  # (makes it in the plot optically more appealing)
  # mutate(sales_text = scales::dollar(sales)) <- Works for dollar values
  mutate(state_text = scales::dollar(sales, big.mark = ".", 
                                     decimal.mark = ",", 
                                     prefix = "", 
                                     suffix = " €"))

14 Step 2 - Visualize

sales_by_year_state %>%

  # Set up x, y, fill
  ggplot(aes(x = year, y = sales, fill = year)) +

  # Geometries
  geom_col() + # Run up to here to get a stacked bar plot

  # Facet
  facet_wrap(~ State) +

  # Formatting
  scale_y_continuous(labels = scales::dollar_format(big.mark = ".", 
                                                    decimal.mark = ",", 
                                                    prefix = "", 
                                                    suffix = " €"))
